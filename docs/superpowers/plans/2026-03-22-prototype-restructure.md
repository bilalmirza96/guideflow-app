# Prototype Restructure Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add marker-based extract/import tooling to the prototype monolith so pages can be edited individually while the monolith remains the source of truth.

**Architecture:** HTML comment markers (`<!-- SECTION:type:page START/END -->`) delimit every extractable section in `guideflow-app.html`. A Node.js CLI (`proto.js`) extracts pages to standalone files, imports edits back, and regenerates all split pages. Missing page-snippets get integrated first.

**Tech Stack:** Node.js (fs, path — no dependencies), HTML/CSS/JS

**Spec:** `docs/superpowers/specs/2026-03-22-prototype-restructure-design.md`

- [ ] **Step 2: Add section extraction helper**

This function extracts content between `SECTION:name START` and `SECTION:name END` markers. It handles both HTML comments (`<!-- -->`), CSS comments (`/* */`), and JS comments (`//`).

```js
/**
 * Extract content between SECTION markers.
 * Supports HTML (<!-- -->), CSS (/* */), and JS (//) comment styles.
 * Returns the content BETWEEN the markers (not including the markers themselves).
 * Returns null if section not found.
 */
function extractSection(src, sectionName) {
  // Try HTML comment style
  const htmlStart = `<!-- SECTION:${sectionName} START -->`;
  const htmlEnd = `<!-- SECTION:${sectionName} END -->`;
  let startIdx = src.indexOf(htmlStart);
  let endIdx = src.indexOf(htmlEnd);

  if (startIdx === -1) {
    // Try CSS comment style
    const cssStart = `/* SECTION:${sectionName} START */`;
    const cssEnd = `/* SECTION:${sectionName} END */`;
    startIdx = src.indexOf(cssStart);
    endIdx = src.indexOf(cssEnd);

    if (startIdx === -1) {
      // Try JS comment style
      const jsStart = `// SECTION:${sectionName} START`;
      const jsEnd = `// SECTION:${sectionName} END`;
      startIdx = src.indexOf(jsStart);
      endIdx = src.indexOf(jsEnd);

      if (startIdx === -1) return null;
      return src.substring(startIdx + jsStart.length, endIdx).trim();
    }
    return src.substring(startIdx + cssStart.length, endIdx).trim();
  }
  return src.substring(startIdx + htmlStart.length, endIdx).trim();
}

/**
 * Replace content between SECTION markers in source.
 * Returns updated source, or null if section not found.
 */
function replaceSection(src, sectionName, newContent) {
  // Try all comment styles
  const patterns = [
    { start: `<!-- SECTION:${sectionName} START -->`, end: `<!-- SECTION:${sectionName} END -->` },
    { start: `/* SECTION:${sectionName} START */`, end: `/* SECTION:${sectionName} END */` },
    { start: `// SECTION:${sectionName} START`, end: `// SECTION:${sectionName} END` },
  ];

  for (const p of patterns) {
    const startIdx = src.indexOf(p.start);
    const endIdx = src.indexOf(p.end);
    if (startIdx !== -1 && endIdx !== -1) {
      return src.substring(0, startIdx + p.start.length) + '\n' + newContent + '\n' + src.substring(endIdx);
    }
  }
  return null;
}
```

```js
function importPage(pageKey) {
  const inFile = path.join(PAGES_DIR, FILENAME_MAP[pageKey]);
  if (!fs.existsSync(inFile)) {
    console.error(`Error: ${inFile} not found`);
    process.exit(1);
  }

  const pageSrc = fs.readFileSync(inFile, 'utf8');
  let monolith = fs.readFileSync(MONOLITH, 'utf8');
  const updated = [];

  // Page-specific sections to import (never import shared sections)
  const sections = [
    `sidebar:${pageKey}`,
    `page:${pageKey}`,
    `styles:${pageKey}`,
    `scripts:${pageKey}`,
    `modals:${pageKey}`,
  ];

  for (const section of sections) {
    const content = extractSection(pageSrc, section);
    if (content === null) continue; // Section not in page file, skip

    const result = replaceSection(monolith, section, content);
    if (result === null) {
      console.warn(`  Warning: SECTION:${section} not found in monolith, skipping`);
      continue;
    }
    monolith = result;
    updated.push(section);
  }

  if (updated.length === 0) {
    console.log(`No sections found to import for ${pageKey}`);
    return;
  }

  fs.writeFileSync(MONOLITH, monolith);
  console.log(`Imported ${pageKey} → updated ${updated.length} sections: ${updated.join(', ')}`);
}
```

```js
function generateSharedAssets() {
  const src = fs.readFileSync(MONOLITH, 'utf8');
  const styles = extractSection(src, 'styles');
  const scripts = extractSection(src, 'scripts');

  if (styles) {
    fs.writeFileSync(
      path.join(PAGES_DIR, 'shared.css'),
      `/* GuideFlow Shared Styles - extracted from guideflow-app.html */\n${styles}`
    );
    console.log('  shared.css');
  }
  if (scripts) {
    // Replace navigate() with window.location.href for multi-page mode
    let multiPageScripts = scripts.replace(
      /function navigate\(page\)\{[\s\S]*?window\.scrollTo\(0,0\)\}/,
      `function navigate(page) {
  var pageMap = ${JSON.stringify(FILENAME_MAP)};
  if (pageMap[page]) window.location.href = pageMap[page];
}`
    );
    fs.writeFileSync(
      path.join(PAGES_DIR, 'shared.js'),
      `// GuideFlow Shared Scripts - extracted from guideflow-app.html\n${multiPageScripts}`
    );
    console.log('  shared.js');
  }
}

function splitAll() {
  for (const key of PAGE_KEYS) {
    extractPage(key);
  }
  generateSharedAssets();
  console.log(`\nDone! All ${PAGE_KEYS.length} pages + shared assets written to prototype/pages/`);
}

// Main dispatch
if (command === 'extract') {
  extractPage(pageArg);
} else if (command === 'import') {
  importPage(pageArg);
} else if (command === 'split') {
  splitAll();
}
```
